178    def watch(self, item):
179        '''Watch local state variable for updating based on a response from the JS8Call application.
180
181        The JS8Call application responds to requests asynchronously, which means responses may not be received in the same order as the requests or in a reasonable timeframe following the request. To handle these asynchronous reponses, the responses set local state variables as they are received. The process is as follows:
182        - Send a request to the JS8Call application
183        - Set local state variable associated with anticipated response to a known value (None)
184        - Wait for a response, timing out if a response is not received within 3 seconds
185        - Detect a change in the local state variable indicating that a response has been received
186        - Return the response (i.e. the current value of the updated local state variable)
187
188        If a timeout occurs while waiting for a response the local state variable is set back to its previous value.
189
190        Args:
191            item (str): Name of the local state variable to watch
192
193        Returns:
194            The value of the given local state variable
195        '''
196        if item not in self.state.keys():
197            return None
198
199        self._watching = item
200        last_state = self.state[item]
201        self.state[item] = None
202        timeout = time.time() + self._watch_timeout
203
204        while timeout > time.time():
205            if self.state[item] != None:
206                break
207            time.sleep(0.001)
208
209        # timeout occurred, revert to last state
210        if self.state[item] == None:
211            self.state[item] = last_state
212        
213        self._watching = None
214        return self.state[item]

216    def spot(self, msg):
217        '''Store a local spot when a station is heard.
218
219        A spot object (see pyjs8call.spot) is constructed and stored. The new spot object is compared against a list of recent spot objects (heard within the last 10 seconds) to prevent duplicate spots from multiple JS8Call API messages associated with the same station event.
220
221        The list of stored spots is culled once it exceeeds the maximum size set by *max_spots* by dropping the oldest spot.
222
223        See pyjs8call.client.Client.get_station_spots to utilize stored spots.
224
225        Args:
226            msg (pyjs8call.message): Message to source spot data from
227        '''
228        # cull recent spots
229        self._recent_spots = [spot for spot in self._recent_spots if spot.age() < 10]
230
231        new_spot = Spot(msg)
232
233        if new_spot not in self._recent_spots:
234            self._recent_spots.append(new_spot)
235            self.spots.append(new_spot)
236
237        # cull spots
238        if len(self.spots) > self.max_spots:
239            self.spots.pop(0)

241    def _hb(self):
242        '''JS8Call application heartbeat thread.
243
244        If no messages have been received from the JS8Call in the last 5 minutes, a request is issued to the application to make sure it is still connected and functioning as expected.
245        '''
246        while self.online:
247            # if no recent rx, check the connection by making a request
248            timeout = self._last_rx_timestamp + self._socket_heartbeat_delay
249            if time.time() > timeout:
250                self.connected = False
251                msg = Message()
252                msg.type = Message.STATION_GET_CALLSIGN
253                self.send(msg)
254                
255            time.sleep(1)

257    def _tx(self):
258        '''JS8Call application transmit thread.
259
260        The JS8Call tx text field is read every second by pyjs8call.txmonitor. This is utilized to reduce additional socket traffic by reading the local state variable directly without requesting additional updates from the application.
261
262        When processing the transmit message queue, if a transmission is in process (i.e. there is text in the tx text field) then transmission of additional messages is prevented until the current transmission is complete.
263
264        If debugging is enabled (see pyjs8call.client.Client.start) then the byte string of each message sent over the TCP socket is printed to the console. By default not all messages are printed in debug mode (see pyjs8call.js8call.JS8Call._debug_type_blacklist). Frequently sent and received messages used internal to pyjs8call are not printed.
265        '''
266        tx_text = False
267        force_tx_text = False
268
269        while self.online:
270            # TxMonitor updates tx_text every second
271            # do not attempt to update while value is being watched (i.e. updated)
272            if self._watching != 'tx_text' and self.state['tx_text'] != None:
273                if len(self.state['tx_text'].strip()) > 0:
274                    tx_text = True
275                    force_tx_text = False
276                else:
277                    tx_text = False
278
279            self._tx_queue_lock.acquire()
280
281            for msg in self._tx_queue.copy():
282
283                # hold off on sending messages while there is something being sent (text in the tx text field)
284                if msg.type == Message.TX_SEND_MESSAGE and (tx_text or force_tx_text):
285                    continue
286                else:
287                    # pack msg
288                    packed = msg.pack()
289
290                    # print packed msg in debug mode
291                    if self._debug:
292                        if self._debug_all:
293                            print('TX: ' + packed.decode('utf-8').strip())
294                        elif msg.type not in self._debug_type_blacklist:
295                            print('TX: ' + packed.decode('utf-8').strip())
296
297                    # send msg via socket
298                    self._socket.sendall(packed)
299                    # remove msg from queue
300                    self._tx_queue.remove(msg)
301                    # make sure the next queued msg doesn't get sent before the tx text state updates
302                    if msg.type == Message.TX_SEND_MESSAGE:
303                        force_tx_text = True
304
305                    time.sleep(0.1)
306
307            self._tx_queue_lock.release()
308            time.sleep(0.1)

310    def _rx(self):
311        '''JS8Call application receive thread.
312
313        A byte string is read from the TCP socket and parsed into a pyjs8call.message. Socket data is discarded in the following cases:
314            - Failure to decode received byte string to UTF-8 (likely due to corrupted or incomplete data)
315            - Length of received byte string is zero (no data received)
316            - Failure to parse received data into a pyjs8call.message (likely due to corrupted or incomplete data)
317            - Parsed message value contains the JS8Call error character (defaults to an ellipsis)
318
319        Received data that is successfully parsed is passed to *_process_message()* for further processing. 
320
321        If debugging is enabled (see pyjs8call.client.Client.start) then the byte string of each message sent over the TCP socket is printed to the console. By default not all messages are printed in debug mode (see pyjs8call.js8call.JS8Call._debug_type_blacklist). Frequently sent and received messages used internal to pyjs8call are not printed.
322        '''
323        while self.online:
324            data = b''
325            data_str = ''
326
327            try:
328                data += self._socket.recv(65535)
329            except (socket.timeout, OSError):
330                # if rx from socket fails, stop processing
331                # OSError occurs while app is restarting
332                continue
333
334            try: 
335                data_str = data.decode('utf-8')
336            except Exception as e:
337                # if decode fails, stop processing
338                continue
339
340            # if rx data is empty, stop processing
341            if len(data_str) == 0:
342                continue
343
344            self._last_rx_timestamp = time.time()
345            self.connected = True
346
347            # split received data into messages
348            msgs = data_str.split('\n')
349
350            for msg_str in msgs:
351                # if message is empty, stop processing
352                if len(msg_str) == 0:
353                    continue
354
355                try:
356                    msg = Message().parse(msg_str)
357                except Exception as e:
358                    # if parsing message fails, stop processing
359                    continue
360
361                # if error in message value, stop processing
362                if msg.value != None and Message.ERR in msg.value:
363                    continue
364
365                # print msg in debug mode
366                if self._debug:
367                    if self._debug_all:
368                        print('RX: ' + str(msg.dict()))
369                    elif msg.type not in self._debug_type_blacklist:
370                        print('RX: ' + str(msg.dict()))
371
372                self._process_message(msg)
373
374        time.sleep(0.1)

376    def _process_message(self, msg):
377        '''Process received message.
378
379        Messages are processed based on their command and/or their type (see pyjs8call.message for commands and types). Messages are spotted when appropriate. Responses to JS8Call setting and state requests are handled to update the associated local state variables.
380
381        Automatic cleaning of directed message text is handled if enabled (see pyjs8call.client).
382
383        Tx frame messages are passed to pyjs8call.windowmonitor.
384
385        Args:
386            msg (pyjs8call.message): Message to process
387        '''
388
389        ### command handling ###
390
391        if msg.cmd == Message.CMD_HEARING:
392            #TODO validate response structure
393            #if not Message.ERR in msg.params['TEXT']:
394            #    hearing = msg.params['TEXT'].split()[3:]
395            #    for station in hearing:
396            #        if station not in self.spots[msg.params['FROM']].keys():
397            #            self.spots[msg.params['FROM']][station] = []
398            #        self.spots[msg.params['FROM']][station].append(msg)
399
400            # spot message
401            self.spot(msg)
402
403        #TODO no example, test response and update code
404        #elif msg.params['CMD'] == 'QUERY CALL':
405        #    # spot message
406        #    self.spot(msg)
407                
408        elif msg.cmd in Message.COMMANDS:
409            # spot message
410            self.spot(msg)
411
412
413        ### message type handling ###
414
415        if msg.type == Message.INBOX_MESSAGES:
416            self.state['inbox'] = msg.messages
417
418        elif msg.type == Message.RX_SPOT:
419            # spot message
420            self.spot(msg)
421
422        elif msg.type == Message.RX_DIRECTED:
423            # clean msg text to remove callsigns, etc
424            if self._client.clean_directed_text:
425                msg = self._client.clean_rx_message_text(msg)
426
427            # spot message
428            self.spot(msg)
429
430        elif msg.type == Message.RIG_FREQ:
431            self.state['dial'] = msg.dial
432            self.state['freq'] = msg.freq
433            self.state['offset'] = msg.offset
434
435        elif msg.type == Message.RIG_PTT:
436            if msg.value == 'on':
437                self.state['ptt'] = True
438            else:
439                self.state['ptt'] = False
440
441        elif msg.type == Message.STATION_STATUS:
442            self.state['dial'] = msg.dial
443            self.state['freq'] = msg.freq
444            self.state['offset'] = msg.offset
445            self.state['speed'] = msg.speed
446
447        elif msg.type == Message.STATION_CALLSIGN:
448            self.state['callsign'] = msg.value
449
450        elif msg.type == Message.STATION_GRID:
451            self.state['grid'] = msg.value
452
453        elif msg.type == Message.STATION_INFO:
454            self.state['info'] = msg.value
455
456        elif msg.type == Message.MODE_SPEED:
457            self.state['speed'] = msg.speed
458
459        elif msg.type == Message.TX_TEXT:
460            self.state['tx_text'] = msg.value
461
462        elif msg.type == Message.RX_TEXT:
463            self.state['rx_text'] = msg.value
464
465        elif msg.type == Message.RX_SELECTED_CALL:
466            self.state['selected_call'] = msg.value
467
468        elif msg.type == Message.RX_CALL_ACTIVITY:
469            self.state['call_activity'] = msg.call_activity
470
471        elif msg.type == Message.RX_BAND_ACTIVITY:
472            self.state['band_activity'] = msg.band_activity
473
474        #TODO should this be used? use RX.BAND_ACTIVITY for now
475        #TODO note, RX.SPOT received immediately after RX.ACTIVITY in some cases
476        elif msg.type == Message.RX_ACTIVITY:
477            pass
478
479        elif msg.type == Message.TX_FRAME:
480            self._client.window_monitor.process_tx_frame(msg)
481
482        self.append_to_rx_queue(msg)

258    def _tx(self):
259        '''JS8Call application transmit thread.
260
261        The JS8Call tx text field is read every second by pyjs8call.txmonitor. This is utilized to reduce additional socket traffic by reading the local state variable directly without requesting additional updates from the application.
262
263        When processing the transmit message queue, if a transmission is in process (i.e. there is text in the tx text field) then transmission of additional messages is prevented until the current transmission is complete.
264
265        If debugging is enabled (see pyjs8call.client.Client.start) then the byte string of each message sent over the TCP socket is printed to the console. By default not all messages are printed in debug mode (see pyjs8call.js8call.JS8Call._debug_type_blacklist). Frequently sent and received messages used internal to pyjs8call are not printed.
266        '''
267        tx_text = False
268        force_tx_text = False
269
270        while self.online:
271            # TxMonitor updates tx_text every second
272            # do not attempt to update while value is being watched (i.e. updated)
273            if self._watching != 'tx_text' and self.state['tx_text'] != None:
274                if len(self.state['tx_text'].strip()) > 0:
275                    tx_text = True
276                    force_tx_text = False
277                else:
278                    tx_text = False
279
280            self._tx_queue_lock.acquire()
281
282            for msg in self._tx_queue.copy():
283
284                # hold off on sending messages while there is something being sent (text in the tx text field)
285                if msg.type == Message.TX_SEND_MESSAGE and (tx_text or force_tx_text):
286                    continue
287                else:
288                    # pack msg
289                    packed = msg.pack()
290
291                    # print packed msg in debug mode
292                    if self._debug:
293                        if self._debug_all:
294                            print('TX: ' + packed.decode('utf-8').strip())
295                        elif msg.type not in self._debug_type_blacklist:
296                            print('TX: ' + packed.decode('utf-8').strip())
297
298                    # send msg via socket
299                    self._socket.sendall(packed)
300                    # remove msg from queue
301                    self._tx_queue.remove(msg)
302                    # make sure the next queued msg doesn't get sent before the tx text state updates
303                    if msg.type == Message.TX_SEND_MESSAGE:
304                        force_tx_text = True
305
306                    time.sleep(0.1)
307
308            self._tx_queue_lock.release()
309            time.sleep(0.1)

311    def _rx(self):
312        '''JS8Call application receive thread.
313
314        A byte string is read from the TCP socket and parsed into a pyjs8call.message. Socket data is discarded in the following cases:
315            - Failure to decode received byte string to UTF-8 (likely due to corrupted or incomplete data)
316            - Length of received byte string is zero (no data received)
317            - Failure to parse received data into a pyjs8call.message (likely due to corrupted or incomplete data)
318            - Parsed message value contains the JS8Call error character (defaults to an ellipsis)
319
320        Received data that is successfully parsed is passed to *_process_message()* for further processing. 
321
322        If debugging is enabled (see pyjs8call.client.Client.start) then the byte string of each message sent over the TCP socket is printed to the console. By default not all messages are printed in debug mode (see pyjs8call.js8call.JS8Call._debug_type_blacklist). Frequently sent and received messages used internal to pyjs8call are not printed.
323        '''
324        while self.online:
325            data = b''
326            data_str = ''
327
328            try:
329                data += self._socket.recv(65535)
330            except (socket.timeout, OSError):
331                # if rx from socket fails, stop processing
332                # OSError occurs while app is restarting
333                continue
334
335            try: 
336                data_str = data.decode('utf-8')
337            except Exception as e:
338                # if decode fails, stop processing
339                continue
340
341            # if rx data is empty, stop processing
342            if len(data_str) == 0:
343                continue
344
345            self._last_rx_timestamp = time.time()
346            self.connected = True
347
348            # split received data into messages
349            msgs = data_str.split('\n')
350
351            for msg_str in msgs:
352                # if message is empty, stop processing
353                if len(msg_str) == 0:
354                    continue
355
356                try:
357                    msg = Message().parse(msg_str)
358                except Exception as e:
359                    # if parsing message fails, stop processing
360                    continue
361
362                # if error in message value, stop processing
363                if msg.value != None and Message.ERR in msg.value:
364                    continue
365
366                # print msg in debug mode
367                if self._debug:
368                    if self._debug_all:
369                        print('RX: ' + str(msg.dict()))
370                    elif msg.type not in self._debug_type_blacklist:
371                        print('RX: ' + str(msg.dict()))
372
373                self._process_message(msg)
374
375        time.sleep(0.1)

180        '''Watch local state variable for updating based on a response from the JS8Call application.
181
182        The JS8Call application responds to requests asynchronously, which means responses may not be received in the same order as the requests or in a reasonable timeframe following the request. To handle these asynchronous reponses, the responses set local state variables as they are received. The process is as follows:
183        - Send a request to the JS8Call application
184        - Set local state variable associated with anticipated response to a known value (None)
185        - Wait for a response, timing out if a response is not received within 3 seconds
186        - Detect a change in the local state variable indicating that a response has been received
187        - Return the response (i.e. the current value of the updated local state variable)
188
189        If a timeout occurs while waiting for a response the local state variable is set back to its previous value.
190
191        Args:
192            item (str): Name of the local state variable to watch
193
194        Returns:
195            The value of the given local state variable
196        '''
197        if item not in self.state.keys():
198            return None
199
200        self._watching = item
201        last_state = self.state[item]
202        self.state[item] = None
203        timeout = time.time() + self._watch_timeout
204
205        while timeout > time.time():
206            if self.state[item] != None:
207                break
208            time.sleep(0.001)
209
210        # timeout occurred, revert to last state
211        if self.state[item] == None:

213
