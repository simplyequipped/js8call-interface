pyjs8call.client

Main JS8Call API interface.

Includes many functions for reading/writing settings and sending various types of messages.

Typical usage example:
js8call = pyjs8call.Client()
js8call.register_rx_callback(rx_func)
js8call.start()

js8call.send_directed_message('KT1RUN', 'Great content thx')

 66    def __init__(self, host='127.0.0.1', port=2442, headless=False, config_path=None):
 67        '''Initialize JS8Call API client.
 68
 69        Registers the Client.stop function with the atexit module.
 70
 71        Args:
 72            host (str): JS8Call TCP address setting, defaults to '127.0.0.1'
 73            port (int): JS8Call TCP port setting, defaults to 2442
 74            headless (bool): Run JS8Call headless using xvfb (linux only, requires xvfb to be installed), defaults to False
 75            config_path (str): Non-standard JS8Call.ini configuration file path, defaults to None
 76
 77        Returns:
 78            pyjs8call.client.Client: Constructed client object
 79        '''
 80        self.host = host
 81        self.port = port
 82        self.headless = headless
 83        self.clean_directed_text = True
 84        self.monitor_directed_tx = True
 85        self.online = False
 86
 87        self.js8call = None
 88        self.spot_monitor = None
 89        self.window_monitor = None
 90        self.offset_monitor = None
 91        self.tx_monitor = None
 92
 93        # delay between setting value and getting updated value
 94        self._set_get_delay = 0.1 # seconds
 95
 96        # initialize the config file handler
 97        self.config = pyjs8call.ConfigHandler(config_path = config_path)
 98
 99        self.callbacks = {
100            Message.RX_DIRECTED: [],
101        }
102
103        # stop application and client at exit
104        atexit.register(self.stop)

106    def set_config_profile(self, profile):
107        '''Set active JS8Call configuration profile in the JS8Call.ini file.
108
109        Restarts the JS8Call client (self) if already online.
110
111        Args:
112            profile (str): Profile name
113
114        Raises:
115            Exception: Given profile name does not exist (see pyjs8call.confighandler.ConfigHandler.create_new_profile)
116        '''
117        if profile not in self.config.get_profile_list():
118            raise Exception('Config profile ' + profile + ' does not exist')
119
120        # set the profile as active
121        self.config.change_profile(profile)
122
123        # restart the app to apply new profile if already running
124        if self.online:
125            self.restart()

127    def start(self, debug=False):
128        '''Start and connect to the the JS8Call application.
129
130        Starts monitoring objects and associated threads:
131            - Spot monitor (see pyjs8call.spotmonitor)
132            - Window monitor (see pyjs8call.windowmonitor)
133            - Offset monitor (see pyjs8call.offsetmonitor)
134            - Tx monitor (see pyjs8call.txmonitor)
135
136        Args:
137            debug (bool): Print rx and tx messages to the console for debugging, defaults to False
138        '''
139        # enable TCP connection
140        self.config.set('Configuration', 'TCPEnabled', 'true')
141        self.config.set('Configuration', 'TCPServer', self.host)
142        self.config.set('Configuration', 'TCPServerPort', str(self.port))
143        self.config.set('Configuration', 'AcceptTCPRequests', 'true')
144        self.config.write()
145
146        # start js8call app and TCP interface
147        self.js8call = pyjs8call.JS8Call(self, self.host, self.port, headless=self.headless)
148        self.online = True
149
150        if debug:
151            self.js8call._debug = True
152
153        # initialize rx thread
154        rx_thread = threading.Thread(target=self._rx)
155        rx_thread.setDaemon(True)
156        rx_thread.start()
157
158        time.sleep(0.5)
159
160        # start station spot monitor
161        self.spot_monitor = pyjs8call.SpotMonitor(self)
162        # start tx window monitor
163        self.window_monitor = pyjs8call.WindowMonitor(self)
164        # start auto offset monitor
165        self.offset_monitor = pyjs8call.OffsetMonitor(self)
166        # start tx monitor
167        self.tx_monitor = pyjs8call.TxMonitor(self)
168
169    def stop(self):
170        '''Stop all threads, close the TCP socket, and kill the JS8Call application.'''
171        self.online = False
172        try:
173            self.js8call.stop()
174        except:
175            pass

177    def restart(self):
178        '''Stop and restart all threads, the JS8Call application, and the TCP socket.
179
180        Dial frequency, offset frequency, and all callback functions are preserved.
181        '''
182        # save callback settings
183        tx_monitor_status_change_callback = self.tx_monitor._status_change_callback
184        spot_monitor_new_spot_callback = self.spot_monitor._new_spot_callback
185        spot_monitor_watch_callback = self.spot_monitor._watch_callback
186        window_monitor_window_callback = self.window_monitor._window_callback
187
188        # save freq and offset
189        freq = self.get_freq()
190        offset = self.get_offset()
191
192        self.stop()
193        self.js8call._socket.close()
194        time.sleep(1)
195        self.start(debug = self.js8call._debug)
196
197        # restore callback settings
198        self.tx_monitor._status_change_callback = tx_monitor_status_change_callback
199        self.spot_monitor._new_spot_callback = spot_monitor_new_spot_callback
200        self.spot_monitor._watch_callback = spot_monitor_watch_callback
201        self.window_monitor._window_callback = window_monitor_window_callback
202
203        # restore freq and offset
204        self.set_offset(offset)
205        self.set_freq(freq)

207    def register_rx_callback(self, callback, message_type=Message.RX_DIRECTED):
208        '''Register a rx callback function.
209
210        Callback functions are associated with specific message types. The directed message type is assumed unless otherwise specified. See pyjs8call.message for specific message types.
211
212        Args:
213            callback (func): Callback function object with the signature func(msg) where msg is a pyjs8call.message object
214            message_type (str): The message type to associate with the callback funtion
215        '''
216        if message_type not in self.callbacks.keys():
217            self.callbacks[message_type] = []
218
219        self.callbacks[message_type].append(callback)
220
221    def _rx(self):
222        '''Rx thread function.'''
223        while self.online:
224            msg = self.js8call.get_next_message()
225
226            if msg != None and msg.type in self.callbacks.keys():
227                for callback in self.callbacks[msg.type]:
228                    callback(msg)
229
230            time.sleep(0.1)

275    def clean_rx_message_text(self, msg):
276        '''Clean rx message text.
277
278        Remove origin callsign, remove destination callsign or group, and strip whitespace and end-of-message characters. This leaves only the message text.
279        The Message.text attribute stores the cleaned text while the Message.value attribute is unchanged.
280
281        Args:
282            message (pyjs8call.message): Message object to clean
283
284        Returns:
285            pyjs8call.message: Cleaned message object
286        '''
287        if msg == None:
288            return None
289        elif msg.value == None or msg.value == '':
290            # nothing to clean
291            return msg
292        # already cleaned
293        elif msg.value != msg.text:
294            return msg
295
296        message = msg.value
297        # remove origin callsign
298        message = message.split(':')[1].strip()
299        # remove destination callsign or group
300        message = ' '.join(message.split(' ')[1:])
301        # strip remaining spaces and end-of-message symbol
302        message = message.strip(' ' + Message.EOM)
303
304        msg.set('text', message)
305        return msg

276    def clean_rx_message_text(self, msg):
277        '''Clean rx message text.
278
279        Remove origin callsign, remove destination callsign or group, and strip whitespace and end-of-message characters. This leaves only the message text.
280        The Message.text attribute stores the cleaned text while the Message.value attribute is unchanged.
281
282        Args:
283            message (pyjs8call.message): Message object to clean
284
285        Returns:
286            pyjs8call.message: Cleaned message object
287        '''
288        if msg == None:
289            return None
290        elif msg.value == None or msg.value == '':
291            # nothing to clean
292            return msg
293        # already cleaned
294        elif msg.value != msg.text:
295            return msg
296
297        message = msg.value
298        # remove origin callsign
299        message = message.split(':')[1].strip()
300        # remove destination callsign or group
301        message = ' '.join(message.split(' ')[1:])
302        # strip remaining spaces and end-of-message symbol
303        message = message.strip(' ' + Message.EOM)
304
305        msg.set('text', message)
