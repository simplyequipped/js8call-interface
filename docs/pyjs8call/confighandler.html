pyjs8call.confighandler

Read from and write to the JS8Call.ini config file.

This module is initialized by pyjs8call.client.

See the JS8Call.ini config file (located at ~/.config/JS8Call.ini on linux systems) for section titles and options. Pay special attention to option spelling and capitalization. Configuring options incorrectly may cause the JS8Call application not to start.

Note that JS8Call will need to be restarted to implement changes made while JS8Call is running.

Typical usage example:
js8call.config.get_profile_list()
js8call.config.create_new_profile('AppName')
js8call.config.change_profile('AppName')
js8call.config.get_active_profile()

js8call.config.change_profile('Default')
js8call.set('Configuration', 'AvoidAllcall', 'true')
js8call.config.set_profile_option('AppName', 'Configuration', 'Miles', 'true')

 55    '''JS8Call.ini configuration file handler.
 56
 57    Attributes:
 58        path (str): File path to the JS8Call config file, defaults to *~/.config/JS8Call.ini*
 59        config (configparser.ConfigParser): Config parser object containing config file data
 60    '''
 61    def __init__(self, config_path=None):
 62        '''Initialize JS8Call config handler.
 63
 64        Args:
 65            config_path (str): Alternate config file path, defaults to None
 66
 67        Raises:
 68            FileNotFoundError: Config file not found at specified path
 69        '''
 70        if config_path == None:
 71            self.path = os.path.join(os.path.expanduser('~'), '.config/JS8Call.ini')
 72        else:
 73            self.path = config_path
 74
 75        if not os.path.exists(self.path):
 76            raise FileNotFoundError('JS8Call config file not found at ' + str(self.path))
 77
 78        self.config = configparser.ConfigParser(interpolation = None)
 79        self.config.optionxform = lambda option: option
 80        self.config.read(self.path)
 81
 82    def write(self):
 83        '''Write the config parser object to file.
 84
 85        A backup of the original JS8Call config file is saved as JS8Call.ini.original in the same directory as the JS8Call.ini file.
 86        '''
 87        if not os.path.exists(self.path + '.original'):
 88            # create a backup of the original config file before writing changes
 89            with open(self.path + '.original', 'w') as fd:
 90                self.config.write(fd, space_around_delimiters = False)
 91
 92        with open(self.path, 'w') as fd:
 93            # write current config object to the config file
 94            self.config.write(fd, space_around_delimiters = False)
 95
 96    def set(self, section, option, value):
 97        '''Set an option value in a given section.
 98
 99        Value can be of the following types:
100            - str
101            - int
102            - float
103            - bool
104
105        Note that config file boolean types are the lower case strings 'true' and 'false'.
106
107        Args:
108            section (str): Section name containing the specified option
109            option (str): Option name to set value for
110            value (str, int, float, bool): Option value to set
111
112        Returns:
113            str: Value of the specified option in the specified section, or None if value is a type other than those listed above
114        '''
115        self.config.set(section, option, str(value))
116        if isinstance(value, (str, int, float, bool)):
117            return self.get(section, option, value_type=type(value))
118        else:
119            return None
120
121    def get(self, section, option, value_type=str(), fallback=None):
122        '''Get an option value from a given section.
123
124        Values can be of the following types:
125            - str
126            - int
127            - float
128            - bool
129
130        Args:
131            section (str): Section name containing the specified option
132            option (str): Option name to get value for
133            value_type (str, int, float, bool): A variable or constructor function of the same type as the value to be returned
134            fallback (any): The value to be returned if the given option is not found, defaults to None
135
136        Returns:
137            The value of the specified option in the specified section as the specified type, or the fallback value if the option is not found.
138        '''
139        if isinstance(value_type, str):
140            return self.config.get(section, option, fallback=fallback)
141        elif isinstance(value_type, int):
142            return self.config.getint(section, option, fallback=fallback)
143        elif isinstance(value_type, float):
144            return self.config.getfloat(section, option, fallback=fallback)
145        elif isinstance(value_type, bool):
146            return self.config.getboolean(section, option, fallback=fallback)
147
148    def clear_call_activity(self):

155    def get_active_profile(self):
156        '''Get active JS8Call configuration profile.
157
158        Returns:
159            str: Name of active configuration profile
160        '''
161        return self.get('MultiSettings', 'CurrentName')
162
163    def get_profile_list(self):
164        '''Get list of existing JS8Call configuration profiles.
165
166        Returns:
167            list: List of profiles in the configuration file
168        '''
169        profiles = []
170        profiles.append(self.get_active_profile())
171
172        for option in self.config.options('MultiSettings'):
173            option_parts = option.split('\\')
174
175            if len(option_parts) == 1:
176                continue
177
178            profile_name = option_parts[0]
179
180            if profile_name not in profiles:
181                profiles.append(profile_name)
182
183        return profiles
184
185    def get_profile_options(self, profile):
186        '''Get all options and values for a configuration profile.
187
188        Args:
189            profile (str): The name of the configuration profile to get options and values for
190
191        Returns:
192            A dictionary of the following structure:
193                dict[section][option] = value
194        '''
195        if profile not in self.get_profile_list():
196            raise Exception('Profile ' + profile + ' does not exist')
197
198        options = {}
199
200        for option, value in self.config.items('MultiSettings'):
201            option_parts = option.split('\\')
202
203            if len(option_parts) == 1:
204                continue
205
206            profile_name = option_parts[0]
207            profile_section = option_parts[1]
208
209            if len(option_parts) > 3:
210                profile_option = '\\'.join(option_parts[2:])
211            else:
212                profile_option = option_parts[2]
213
214            if profile_name == profile:
215                if profile_section not in options.keys():
216                    options[profile_section] = {}
217
218                options[profile_section][profile_option] = value
219
220        return options
221
222    def get_profile_option(self, profile, section, option, value_type=str(), fallback=None):
223        '''Get an option value from a given section in a given profile.
224
225        Values can be of the following types:
226            - str
227            - int
228            - float
229            - bool
230
231        Args:
232            profile (str): Profile name containing the the specified section and option
233            section (str): Section name containing the specified option
234            option (str): Option name to get value for
235            value_type (str, int, float, bool): A variable or constructor function of the same type as the value to be returned
236            fallback (any): The value to be returned if the given option is not found, defaults to None
237
238        Returns:
239            The value of the specified option in the specified section in the specified profile as the specified type, or the fallback value if the option is not found.
240        '''
241        if profile not in self.get_profile_list():
242            raise Exception('Profile ' + profile + ' does not exist')
243
244        option = profile + '\\' + section + '\\' + option
245        section = 'MultiSettings'
246        return self.get(section, option, value_type=value_type, fallback=fallback)
247

275    def change_profile(self, new_profile):
276        '''Change JS8Call active configuration profile.
277
278        Args:
279            new_profile (str): Name of the profile to change to
280
281        Raises:
282            Exception: Specified profile does not exist
283        '''
284        if new_profile not in self.get_profile_list():
285            raise Exception('Profile ' + new_profile + ' does not exist')
286
287        active_profile = self.get_active_profile()
288        new_profile_options = self.get_profile_options(new_profile)
289
290        for section in self.config.sections():
291            if section == 'MultiSettings':
292                continue
293
294            for option, value in self.config.items(section):
295                # save setting from currently active profile
296                active_profile_option = active_profile + '\\' + section + '\\' + option
297                self.config.set('MultiSettings', active_profile_option, value)
298
299                # set new setting from newly selected profile
300                new_value = self.get_profile_option(new_profile, section, option)
301                self.config.set(section, option, str(new_value))
302                
303                # remove newly selected profile multisettings
304                new_profile_option = new_profile + '\\' + section + '\\' + option
305                self.config.remove_option('MultiSettings', new_profile_option)
306
307        self.set('MultiSettings', 'CurrentName', new_profile)

343    def remove_profile(self, profile):
344        '''Remove an existing JS8Call configuration profile.
345
346        Args:
347            profile (str): Name of existing configuration profile to remove
348
349        Raises:
350            Exception: Specified profile does not exist
351        '''
352        if profile not in self.get_profile_list():
353            raise Exception('Profile ' + profile + ' does not exist')
354
355        profile_options = self.get_profile_options(profile)
356
357        for section in profile_options.keys():
358            for option in profile_options[section].keys():
359                profile_option = profile + '\\' + section + '\\' + option
360                self.config.remove_option('MultiSettings', profile_option)

362    def get_groups(self):
363        '''Get list of JS8Call callsign groups.
364
365        Returns:
366            list: List of callsign groups
367        '''
368        groups = self.config.get('Configuration', 'MyGroups')
369        groups = groups.split(',')
370        # strip spaces, ensure a single @ symbol
371        groups = ['@' + group.strip(' @') for group in groups if len(group.strip()) > 0]
372        
373        return groups
374
375    def add_group(self, group):
376        '''Add a new JS8Call callsign group.
377
378        Args:
379            group (str): Name of the callsign group to create
380        '''
381        # strip spaces, ensure a single @ symbol
382        group = '@' + group.strip(' @')
383
384        if group not in self.get_groups():
385            groups = self.config.get('Configuration', 'MyGroups')
386            # add second @ symbol to match config file formatting
387            groups += ', @' + group
388            self.config.set('Configuration', 'MyGroups', groups)
389
390    def remove_group(self, group):
391        '''Remove an existing JS8Call callsign group.
392
393        Args:
394            group (str): Name of the callsign group to remove
395        '''
396        # strip spaces, ensure a single @ symbol
397        remove_group = '@' + group.strip(' @')
398        groups = self.get_groups()
399
400        if remove_group in groups:
401            # add second @ symbol to match config file formatting
402            groups = ['@' + group for group in groups if group != remove_group]
403            groups = ', '.join(groups)
404            self.config.set('Configuration', 'MyGroups', groups)
